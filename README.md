Algoritmo "troco"



Var
   preco, dinheiro, troco : real
   quantidade : inteiro
   
 
 Inicio
  escreva("preco unitario do produto: ")
  leia(preco)
  escreva ("quantidade comprada: ")
  leia(quantidade)
  escreva("dinheiro recebido: ")
  leia(dinheiro)
  troco <- dinheiro - (preco * quantidade)
  escreval ("troco = ", troco:8:2)

  






Fimalgoritmo
